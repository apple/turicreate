70 template<class T, class... Ts> void fmt_ext_loop(
71  ostream& os, const string& fmt_str, int64_t i, T arg, Ts... args) {
72 
73  int64_t i_next = 0;
74  while (true) {
75  i_next = fmt_str.find("%", i);
76  if (i_next == -1) {
77  os << fmt_str.substr(i);
78  break;
79  }
80  os << fmt_str.substr(i, i_next - i);
81  i = i_next + 1;
82  if (i >= static_cast<int64_t>(fmt_str.length())) {
83  {
84  ostringstream os_inner;
85  os_inner << " ***** Incomplete pattern at end of format string: " << fmt_str << endl;
86  cerr << os_inner.str();
87  AU();
88  }
89  }
90  if (fmt_str[i] == '%') {
91  os << '%';
92  ++i;
93  continue;
94  }
95  fmt_single(os, fmt_str[i], arg);
96  fmt_ext_loop(os, fmt_str, i+1, args...);
97  break;
98  }
99 }
