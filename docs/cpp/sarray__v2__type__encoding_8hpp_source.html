129 /**
130  * Encodes a collection of flexible_type values. The array must be of
131  * contiguous type, but permitting undefined values.
132  *
133  * There is a two byte header to the block.
134  * - num_types: 1 byte
135  * - if 0, the block is empty.
136  * - if 1, the array is of contiguous type (see next byte)
137  * - if 2, the array is of contiguous type, but has missing values.
138  * - type: 1 byte.
139  * - [undefined bitfield]: if type is 2, this contains a bitfield of
140  * (round_op(#elem / 8) bytes) listing the positions of all the UNDEFINED
141  * fields)
142  * - type specific encoding:
143  * - if integer or float, encode_number() is called
144  * - if string, encode_string() is called
145  * - otherwise, direct serialization is currently used.
146  * - If UNDEFINED (i.e. array is of all UNDEFINED values, nothing is written)
147  *
148  * \note The coding does not store the number of values stored. This is
149  * stored in the block_info (block.num_elem)
150  */

159 template <typename Fn> // Fn is a function like void(flexible_type)
160 static void decode_number_stream(size_t num_elements,
161  iarchive& iarc,
162  Fn callback) {
163  uint64_t buf[MAX_INTEGERS_PER_BLOCK];
164  while(num_elements > 0) {
165  size_t buflen = std::min<size_t>(num_elements, MAX_INTEGERS_PER_BLOCK);
166  frame_of_reference_decode_128(iarc, buflen, buf);
167  for (size_t i = 0;i < buflen; ++i) {
168  callback(flexible_type(buf[i]));
169  }
170  num_elements -= buflen;
171  }
172 }

178 template <typename Fn> // Fn is a function like void(flexible_type)
179 static void decode_double_stream_legacy(size_t num_elements,
180  iarchive& iarc,
181  Fn callback) {
182  uint64_t buf[MAX_INTEGERS_PER_BLOCK];
183  while(num_elements > 0) {
184  size_t buflen = std::min<size_t>(num_elements, MAX_INTEGERS_PER_BLOCK);
185  frame_of_reference_decode_128(iarc, buflen, buf);
186  for (size_t i = 0;i < buflen; ++i) {
187  size_t intval = (buf[i] >> 1) | (buf[i] << 63);
188  // make a double flexible_type
189  flexible_type ret(0.0);
190  ret.reinterpret_mutable_get<flex_int>() = intval;
191  callback(ret);
192  }
193  num_elements -= buflen;
194  }
195 }

201 template <typename Fn> // Fn is a function like void(flexible_type)
202 static void decode_double_stream(size_t num_elements,
203  iarchive& iarc,
204  Fn callback) {
205  // we reserve one character so we can add new encoders as needed in the future
206  char reserved = 0;
207  iarc.read(&(reserved), sizeof(reserved));
208  ASSERT_LT(reserved, 3);
209  if (reserved == DOUBLE_RESERVED_FLAGS::LEGACY_ENCODING) {
210  decode_double_stream_legacy(num_elements, iarc, callback);
211  return;
212  } else if (reserved == DOUBLE_RESERVED_FLAGS::INTEGER_ENCODING) {
213  decode_number_stream(num_elements, iarc,
214  [=](const flexible_type& val) {
215  flex_float ret = flex_float(val.get<flex_int>());
216  callback(ret);
217  });
218  }
219 }

222 /**
223  * Decodes num_elements of strings , calling the callback for each string.

225 template <typename Fn> // Fn is a function like void(flexible_type)
226 static void decode_string_stream(size_t num_elements,
227  iarchive& iarc,
228  Fn callback) {
229  bool use_dictionary_encoding = false;
230  std::vector<flexible_type> idx_values;
231  idx_values.resize(num_elements, flexible_type(flex_type_enum::INTEGER));
232  iarc >> use_dictionary_encoding;
233  if (use_dictionary_encoding) {
234  uint64_t num_values;
235  std::vector<flexible_type> str_values;
236  variable_decode(iarc, num_values);
237  str_values.resize(num_values);
238  for (auto& str: str_values) {
239  std::string new_str;
240  uint64_t str_len;
241  variable_decode(iarc, str_len);
242  new_str.resize(str_len);
243  iarc.read(&(new_str[0]), str_len);
244  str = std::move(new_str);
245  }
246  decode_number(iarc, idx_values, 0);
247  for (size_t i = 0;i < num_elements; ++i) {
248  callback(str_values[idx_values[i].get<flex_int>()]);
249  }
250  } else {
251  // get all the lengths
252  decode_number(iarc, idx_values, 0);
253  flexible_type ret(flex_type_enum::STRING);
254  for (size_t i = 0;i < num_elements; ++i) {
255  size_t str_len = idx_values[i].get<flex_int>();
256  ret.mutable_get<std::string>().resize(str_len);
257  iarc.read(&(ret.mutable_get<std::string>()[0]), str_len);
258  callback(ret);
259  }
260  }
261 }

263 /**
264  * Decodes num_elements of vectors, calling the callback for each string.
265  *
266  * This is the 2nd generation vector decoder. its use is flagged by
267  * turning on the block flag BLOCK_ENCODING_EXTENSION.
268  */
269 template <typename Fn> // Fn is a function like void(flexible_type)
270 static void decode_vector_stream(size_t num_elements,
271  iarchive& iarc,
272  Fn callback,
273  bool new_format) {
274  // we reserve one character so we can add new encoders as needed in the future
275  if (new_format) {
276  char reserved = 0;
277  iarc.read(&(reserved), sizeof(reserved));
278  }
279  // decode the length of each vector
280  std::vector<flexible_type> lengths(num_elements);
281  decode_number(iarc, lengths, 0);
282  size_t total_num_values = 0;
283  for (const flexible_type& length : lengths) {
284  total_num_values += length.get<flex_int>();
285  }
286 
287  // decode the values
288  std::vector<flexible_type> values(total_num_values);
289  if (new_format) {
290  decode_double(iarc, values, 0);
291  } else {
292  decode_double_legacy(iarc, values, 0);
293  }
294 
295  size_t length_ctr = 0;
296  size_t value_ctr = 0;
297  flexible_type ret(flex_type_enum::VECTOR);
298  for (size_t i = 0 ;i < num_elements; ++i) {
299  flex_vec& output_vec = ret.mutable_get<flex_vec>();
300  // resize this to the appropriate length
301  output_vec.resize(lengths[length_ctr].get<flex_int>());
302  ++length_ctr;
303  // fill in the value
304  for(size_t j = 0; j < output_vec.size(); ++j) {
305  output_vec[j] = values[value_ctr].reinterpret_get<flex_float>();
306  ++value_ctr;
307  }
308  callback(ret);
309  }
310 }

313  * Decodes num_elements of nd_vectors, calling the callback for each string.
314  */
315 template <typename Fn> // Fn is a function like void(flexible_type)
316 static void decode_nd_vector_stream(size_t num_elements,
317  iarchive& iarc,
318  Fn callback,
319  bool new_format) {
320  // new_format is ignored. it should always be true.
321  // one character is reserved so we can add new encoders as needed in the future
322  char reserved = 0;
323  iarc.read(&(reserved), sizeof(reserved));
324 
325  std::vector<flexible_type> shape_lengths(num_elements);
326  std::vector<flexible_type> numel(num_elements);
327  std::vector<flexible_type> shapes;
328  std::vector<flexible_type> strides;
329  std::vector<flexible_type> values;
330 
331  // decode shape lengths and numel
332  decode_number(iarc, shape_lengths, 0);
333  decode_number(iarc, numel, 0);
334 
335  // compute the length of shapes and strides
336  size_t sum_shape_len = 0;
337  for (auto i : shape_lengths) sum_shape_len += i.get<flex_int>();
338  // decode shape and strides
339  shapes.resize(sum_shape_len);
340  strides.resize(sum_shape_len);
341  decode_number(iarc, shapes, 0);
342  decode_number(iarc, strides, 0);
343 
344  // compute the length of values
345  size_t sum_values_len = 0;
346  for (auto i : numel) sum_values_len += i.get<flex_int>();
347  values.resize(sum_values_len);
348  decode_double(iarc, values, 0);
349 
350  // emit
351  size_t shape_stride_ctr = 0;
352  size_t value_ctr = 0;
353 
354  std::vector<size_t> ret_shape;
355  std::vector<size_t> ret_stride;
356 
357  for (size_t i = 0 ;i < num_elements; ++i) {
358  // construct the shape and stride
359  ret_shape.resize(shape_lengths[i]);
360  ret_stride.resize(shape_lengths[i]);
361  for (size_t j = 0;j < shape_lengths[i]; ++j) {
362  ret_shape[j] = shapes[shape_stride_ctr].get<flex_int>();
363  ret_stride[j] = strides[shape_stride_ctr].get<flex_int>();
364  ++shape_stride_ctr;
365  }
366 
367  // construct the values
368  size_t ret_numel = numel[i].get<flex_int>();
369  auto ret_values = std::make_shared<flex_nd_vec::container_type>(ret_numel);
370  for (size_t i = 0;i < ret_numel; ++i) {
371  (*ret_values)[i] = values[i + value_ctr].reinterpret_get<flex_float>();
372  }
373  value_ctr += ret_numel;
374  flexible_type ret(flex_nd_vec(ret_values, ret_shape, ret_stride));
375  callback(ret);
376  }
377 }

383 /**
384  * Decodes a collection of flexible_type values. The array must be of
385  * contiguous type, but permitting undefined values.
386  *
387  * See \ref typed_encode() for details.
388  *
389  * \note The coding does not store the number of values stored. This is
390  * stored in the block_info (block.num_elem)
391  */
392 template <typename Fn> // Fn is a function like void(flexible_type)
393 static bool typed_decode_stream_callback(const block_info& info,
394  char* start, size_t len,
395  Fn callback) {
396  if (!(info.flags & IS_FLEXIBLE_TYPE)) {
397  logstream(LOG_ERROR) << "Attempting to decode a non-typed block"
398  << std::endl;
399  return false;
400  }
401  turi::iarchive iarc(start, len);
402 
403  // some basic block properties which will be filled in
404  // number of elements
405  size_t dsize = info.num_elem;
406  // column type
407  flex_type_enum column_type;
408  // num undefined.
409  size_t num_undefined = 0;
410  // undefined bitmap mapping out where are the undefined values.
411  // only specified if num_undefined > 0
412  turi::dense_bitset undefined_bitmap;
413 
414  char num_types;
415  iarc >> num_types;
416  // if it is a multiple type block, we don't perform a type decode
417  bool perform_type_decoding = !(info.flags & MULTIPLE_TYPE_BLOCK);
418  if (perform_type_decoding) {
419  if (num_types == 0) {
420  // empty block
421  return true;
422  } else if (num_types == 1) {
423  // one block of contiguous type.
424  char c;
425  iarc >> c;
426  column_type = (flex_type_enum)c;
427  // all undefined. generate and return
428  if (column_type == flex_type_enum::UNDEFINED) {
429  for (size_t i = 0;i < dsize; ++i) callback(FLEX_UNDEFINED);
430  return true;
431  }
432  } else if (num_types == 2) {
433  // two types, but with undefined entries.
434  char c;
435  iarc >> c;
436  column_type = (flex_type_enum)c;
437  // read the bitset and undefine all the flagged entries
438  undefined_bitmap.resize(info.num_elem);
439  undefined_bitmap.clear();
440  iarc.read((char*)undefined_bitmap.array, sizeof(size_t)*undefined_bitmap.arrlen);
441  num_undefined = undefined_bitmap.popcount();
442  }
443  } else {
444  std::vector<flexible_type> values;
445  iarc >> values;
446  for (const auto& i: values) {
447  callback(i);
448  }
449  }
450 
451  if (perform_type_decoding) {
452  int last_id = 0;
453  auto stream_callback =
454  [&](const flexible_type& val) {
455  // generate all the undefined
456  if (num_undefined) {
457  while(last_id < dsize &&
458  undefined_bitmap.get(last_id)) {
459  callback(FLEX_UNDEFINED);
460  ++last_id;
461  }
462  }
463  callback(val);
464  ++last_id;
465  };
466  size_t elements_to_decode = dsize - num_undefined;
467  if (column_type == flex_type_enum::INTEGER) {
468  decode_number_stream(elements_to_decode, iarc, stream_callback);
469  } else if (column_type == flex_type_enum::FLOAT) {
470  if (info.flags & BLOCK_ENCODING_EXTENSION) {
471  decode_double_stream(elements_to_decode, iarc, stream_callback);
472  } else {
473  decode_double_stream_legacy(elements_to_decode, iarc, stream_callback);
474  }
475  } else if (column_type == flex_type_enum::STRING) {
476  decode_string_stream(elements_to_decode, iarc, stream_callback);
477  } else if (column_type == flex_type_enum::VECTOR) {
478  decode_vector_stream(elements_to_decode, iarc, stream_callback,
479  info.flags & BLOCK_ENCODING_EXTENSION);
480  } else if (column_type == flex_type_enum::ND_VECTOR) {
481  decode_nd_vector_stream(elements_to_decode, iarc, stream_callback,
482  info.flags & BLOCK_ENCODING_EXTENSION);
483  } else {
484  flexible_type_impl::deserializer s{iarc};
485  flexible_type ret(column_type);
486  for (size_t i = 0;i < dsize; ++i) {
487  if (num_undefined && undefined_bitmap.get(i)) {
488  callback(FLEX_UNDEFINED);
489  } else {
490  ret.apply_mutating_visitor(s);
491  callback(ret);
492  }
493  }
494  }
495  // generate the final undefined values
496  if (num_undefined) {
497  while(last_id < dsize &&
498  undefined_bitmap.get(last_id)) {
499  callback(FLEX_UNDEFINED);
500  ++last_id;
501  }
502  }
503  }
504  return true;
505 }
