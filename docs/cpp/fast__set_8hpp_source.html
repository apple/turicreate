45 template<typename T>
46 void radix_sort(T* array, int offset, int end, int shift) {
47  int x, y;
48  T value, temp;
49  int last[256] = { 0 }, pointer[256];
50 
51  for (x=offset; x<end; ++x) {
52  ++last[(array[x] >> shift) & 0xFF];
53  }
54 
55  last[0] += offset;
56  pointer[0] = offset;
57  for (x=1; x<256; ++x) {
58  pointer[x] = last[x-1];
59  last[x] += last[x-1];
60  }
61 
62  for (x=0; x<256; ++x) {
63  while (pointer[x] != last[x]) {
64  value = array[pointer[x]];
65  y = (value >> shift) & 0xFF;
66  while (x != y) {
67  temp = array[pointer[y]];
68  array[pointer[y]++] = value;
69  value = temp;
70  y = (value >> shift) & 0xFF;
71  }
72  array[pointer[x]++] = value;
73  }
74  }
75 
76  if (shift > 0) {
77  shift -= 8;
78  for (x=0; x<256; ++x) {
79  temp = x > 0 ? pointer[x] - pointer[x-1] : pointer[0] - offset;
80  if (temp > 64) {
81  radix_sort(array, pointer[x] - temp, pointer[x], shift);
82  } else if (temp > 1) {
83  std::sort(array + (pointer[x] - temp), array + pointer[x]);
84  }
85  }
86  }
87 }

90 // A fast set storing values in either a vector of sorted VIDs
91 // or a hash set (cuckoo hash).
92 // If the number of elements is greater than HASH_THRESHOLD,
93 // the hash set is used. Otherwise the vector is used.
94 template<typename T>
95 struct fast_set {
96 
97  const static size_t HASH_THRESHOLD = 64;
98 
99  std::vector<T> vec;
100  turi::cuckoo_set_pow2<T, 3> *cset;
101  fast_set(): cset(NULL) { }
102  fast_set(const fast_set& v):cset(NULL) {
103  (*this) = v;
104  }
105 
106  fast_set& operator=(const fast_set& v) {
107  if (this == &v) return *this;
108  vec = v.vec;
109  if (v.cset != NULL) {
110  // allocate the cuckoo set if the other side is using a cuckoo set
111  // or clear if I alrady have one
112  if (cset == NULL) {
113  cset = new turi::cuckoo_set_pow2<T, 3>(-1, 0, 2 * v.cset->size());
114  }
115  else {
116  cset->clear();
117  }
118  (*cset) = *(v.cset);
119  }
120  else {
121  // if the other side is not using a cuckoo set, lets not use a cuckoo set
122  // either
123  if (cset != NULL) {
124  delete cset;
125  cset = NULL;
126  }
127  }
128  return *this;
129  }
130 
131  ~fast_set() {
132  if (cset != NULL) delete cset;
133  }
134 
135  // assigns a vector of vertex IDs to this storage.
136  // this function will clear the contents of the fast_set
137  // and reconstruct it.
138  // If the assigned values has length >= HASH_THRESHOLD,
139  // we will allocate a cuckoo set to store it. Otherwise,
140  // we just store a sorted vector
141  void assign(const std::vector<T>& _vec) {
142  clear();
143  if (_vec.size() >= HASH_THRESHOLD) {
144  // move to cset
145  cset = new turi::cuckoo_set_pow2<T, 3>(-1, 0, 2 * _vec.size());
146  for(T& v: _vec) {
147  cset->insert(v);
148  }
149  }
150  else {
151  vec = _vec;
152  if (vec.size() > 64) {
153  radix_sort(&(vec[0]), 0, vec.size(), 24);
154  }
155  else {
156  std::sort(vec.begin(), vec.end());
157  }
158  typename std::vector<T>::iterator new_end = std::unique(vec.begin(), vec.end());
159  vec.erase(new_end, vec.end());
160  }
161  }
162 
163  void clear() {
164  vec.clear();
165  if (cset != NULL) {
166  delete cset;
167  cset = NULL;
168  }
169  }
170 
171  size_t size() const {
172  return cset == NULL ? vec.size() : cset->size();
173  }
174 };

206  * Computes the size of the intersection of two fast_set's
207  */
208 template<typename T>
209 static uint32_t count_set_intersect(
210  const fast_set<T>& smaller_set,
211  const fast_set<T>& larger_set) {
212  if (smaller_set.size() > larger_set.size()) {
213  return count_set_intersect(larger_set, smaller_set);
214  }
215  if (smaller_set.cset == NULL && larger_set.cset == NULL) {
216  size_t i = 0;
217  counting_inserter<T> iter(&i);
218  std::set_intersection(smaller_set.vid_vec.begin(), smaller_set.vid_vec.end(),
219  larger_set.vid_vec.begin(), larger_set.vid_vec.end(),
220  iter);
221  return i;
222  }
223  else if (smaller_set.cset == NULL && larger_set.cset != NULL) {
224  size_t i = 0;
225  for (T& value : smaller_set.vec) {
226  i += larger_set.cset->count(value);
227  }
228  return i;
229  }
230  else if (smaller_set.cset != NULL && larger_set.cset == NULL) {
231  size_t i = 0;
232  for (T& value : larger_set.vec) {
233  i += smaller_set.cset->count(value);
234  }
235  return i;
236  }
237  else {
238  size_t i = 0;
239  for (T& value: *(smaller_set.cset)) {
240  i += larger_set.cset->count(value);
241  }
242  return i;
243  }
244 }
