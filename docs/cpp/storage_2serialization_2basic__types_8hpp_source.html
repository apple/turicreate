27  /** Serialization of null terminated const char* strings.
28  * This is necessary to serialize constant strings like
29  * \code
30  * oarc << "hello world";
31  * \endcode
32  */
33  template <typename OutArcType>
34  struct serialize_impl<OutArcType, const char*, false> {
35  static void exec(OutArcType& oarc, const char* const& s) {
36  // save the length
37  // ++ for the \0
38  size_t length = strlen(s); length++;
39  oarc << length;
40  oarc.write(reinterpret_cast<const char*>(s), length);
41  DASSERT_FALSE(oarc.fail());
42  }
43  };

73  /// Deserialization of null terminated char* strings
74  template <typename InArcType>
75  struct deserialize_impl<InArcType, char*, false> {
76  static void exec(InArcType& iarc, char*& s) {
77  // Save the length and check if lengths match
78  size_t length;
79  iarc >> length;
80  s = new char[length];
81  //operator>> the rest
82  iarc.read(reinterpret_cast<char*>(s), length);
83  DASSERT_FALSE(iarc.fail());
84  }
85  };

87  /// Deserialization of fixed length char arrays

90  static void exec(InArcType& iarc, char s[len]) {
91  size_t length;
92  iarc >> length;

101  /// Serialization of std::string
102  template <typename OutArcType>
103  struct serialize_impl<OutArcType, std::string, false> {
104  static void exec(OutArcType& oarc, const std::string& s) {
105  size_t length = s.length();
106  oarc << length;
107  oarc.write(reinterpret_cast<const char*>(s.c_str()),
108  (std::streamsize)length);
109  DASSERT_FALSE(oarc.fail());
110  }

114  /// Deserialization of std::string
115  template <typename InArcType>
116  struct deserialize_impl<InArcType, std::string, false> {
117  static void exec(InArcType& iarc, std::string& s) {
118  //read the length
119  size_t length;
120  iarc >> length;
121  //resize the string and read the characters
122  s.resize(length);
123  iarc.read(&(s[0]), (std::streamsize)length);
124  DASSERT_FALSE(iarc.fail());
125  }
126  };
