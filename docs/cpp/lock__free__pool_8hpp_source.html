16  template <typename T, typename index_type = uint32_t>
17  class lock_free_pool{
18  private:
19  std::vector<T> data;
20  T* lower_ptrlimit;
21  T* upper_ptrlimit;
22  // freelist[i] points to the next free list element
23  // if freelist[i] == index_type(-1), then it is the last element
24  // allocated entries are set to index_type(0), though
25  // note that there is no way to disambiguate between allocated
26  // and non-allocated entries by simply looking at the freelist
27  std::vector<index_type> freelist;
28  typedef lock_free_internal::reference_with_counter<index_type> queue_ref_type;
29  volatile queue_ref_type freelisthead;

31  public:
32  lock_free_pool(size_t poolsize = 0) { reset_pool(poolsize); }

35  void reset_pool(size_t poolsize) {
36  if (poolsize == 0) {
37  data.clear();
38  freelist.clear();
39  lower_ptrlimit = NULL;
40  upper_ptrlimit = NULL;
41  } else {
42  data.resize(poolsize);
43  freelist.resize(poolsize);
44  for (index_type i = 0;i < freelist.size(); ++i) {
45  freelist[i] = i + 1;
46  }
47  freelist[freelist.size() - 1] = index_type(-1);
48  lower_ptrlimit = &(data[0]);
49  upper_ptrlimit = &(data[data.size() - 1]);
50  }
51  freelisthead.q.val = 0;
52  freelisthead.q.counter = 0;
53  }

55  std::vector<T>& unsafe_get_pool_ref() { return data; }

57  T* alloc() {
58  // I need to atomically advance freelisthead to the freelist[head]
59  queue_ref_type oldhead;
60  queue_ref_type newhead;
61  do {
62  oldhead.combined = freelisthead.combined;
63  if (oldhead.q.val == index_type(-1)) return new T; // ran out of pool elements
64  newhead.q.val = freelist[oldhead.q.val];
65  newhead.q.counter = oldhead.q.counter + 1;
66  } while(!atomic_compare_and_swap(freelisthead.combined,
67  oldhead.combined,
68  newhead.combined));
69  freelist[oldhead.q.val] = index_type(-1);
70  return &(data[oldhead.q.val]);
71  }

73  void free(T* p) {
74  // is this from the pool?
75  // if it is below the pointer limits
76  if (__unlikely__(p < lower_ptrlimit || p > upper_ptrlimit)) {
77  delete p;
78  return;
79  }
80 
81  index_type cur = index_type(p - &(data[0]));
82 
83  // prepare for free list insertion
84  // I need to atomically set freelisthead == cur
85  // and freelist[cur] = freelisthead
86  queue_ref_type oldhead;
87  queue_ref_type newhead;
88  do{
89  oldhead.combined = freelisthead.combined;
90  freelist[cur] = oldhead.q.val;
91  newhead.q.val = cur;
92  newhead.q.counter = oldhead.q.counter + 1;
93  // now try to atomically move freelisthead
94  } while(!atomic_compare_and_swap(freelisthead.combined,
95  oldhead.combined,
96  newhead.combined));
97  }
98  }; // end of lock free pool
