23 /**
24  * A buffered reader reading from a range of an sarray<T>.
25  *
26  * \code
27  * sarray<flexible_type> mysarray = ...;
28  *
29  * // Reader for the first thousand lines
30  * sarray_reader_buffer<flexible_type> reader(mysarray, 0, 1000);
31  *
32  * while(reader.has_next()) {
33  * flexible_type val = reader.next();
34  * ... do some thing with val ...
35  * }
36  *
37  * // Reader for the entire sarray
38  * reader = sarray_reader_buffer<flexible_type>(mysarray, 0, (size_t)(-1));
39  * ...
40  * \endcode
41  *
42  * Internally, the reader maintains a vector as buffer, and when reading
43  * reaches the end of the buffer, refill the buffer by reading from sarray.
44  */

60  void init(std::shared_ptr<typename sarray<T>::reader_type>& reader,

116 /// Return the next element in the chunk.
117 template<typename T>
118 T&& sarray_reader_buffer<T>::next() {
119  if (m_buffer_pos == m_buffer.size()) {
120  refill();
121  m_buffer_pos = 0;
122  }
123  DASSERT_LT(m_buffer_pos, m_buffer.size());
124  ++m_iter;
125  return std::move(m_buffer[m_buffer_pos++]);
126 }

136 /// Refill the chunk buffer form the sarray reader.
137 template<typename T>
138 void sarray_reader_buffer<T>::refill() {
139  size_t size_of_refill = std::min<size_t>(m_row_end - m_row_start, m_buffer_size);
140  m_reader->read_rows(m_row_start, m_row_start + size_of_refill, m_buffer);
141  m_row_start += size_of_refill;
142 }
