146  //! insert a range of elements into this set
147  inline void insert(const T* begin, const T* end) {
148  // Ensure that other size is not negative
149  ASSERT_LE(begin, end);
150  // Ensure that the other set has an appropriate size
151  const size_t other_size = end - begin;
152  ASSERT_LE(other_size, MAX_DIM);
153  // Construct a temporary small set representing the range
154  small_set other;
155  for(size_t i = 0; i < other_size; ++i) {
156  other[i] = begin[i];
157  // Ensure that the other set is sorted
158  if(i+1 < other_size) ASSERT_LT(begin[i], begin[i+1]);
159  }
160  // Insert it into this small set using the + operation
161  *this += other;
162  }

241  //! used frequently
242  inline small_set& operator+=(const T& elem) {
243  // // Find where elem should be inserted
244  // size_t index = 0;
245  // for(; index < nelems && values[index] < elem; ++index);
246  T* ptr(std::lower_bound(begin(), end(), elem, Less()));
247  // if the element already exists return
248  if(ptr != end() && !(elem < *ptr) ) return *this;
249  // otherwise the element does not exist so add it at the current
250  // location and increment the number of elements
251  T tmp(elem); nelems++; // advances end
252  ASSERT_LE(nelems, MAX_DIM);
253  // Insert the element at index swapping out the rest of the
254  // array
255  for(; ptr < end(); ++ptr) std::swap(*ptr, tmp);
256  // Finished return
257  return *this;
