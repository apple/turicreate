24 /**
25  * \internal
26  * \ingroup sframe_physical
27  * \addtogroup sframe_internal SFrame Internal
28  *
29  * A more flexible sframe creator.
30  *
31  * User can finely control the ratio of certain categories
32  * of data. For instance, 70% of 1s and 30% of 0s.
33  *
34  * One can achieve that goal by recording the count of already
35  * generated data for each category inside of the functors.
36  *
37  * \param[in] column_names: column names
38  * \param[in] column_types: column types
39  * \param[in] nrows: number of rows.
40  *
41  * \param[in] next_row: callable, equivalent to
42  * `std::function<std::vector<flexible_type>(size_t)>`.
43  * if next_row is a function, it should be thread-safe.
44  * if next_row is a functor, it's not required to be thread-safe.
45  */
46 
47 template <typename Callable>
48 sframe make_testing_sframe(const std::vector<std::string>& column_names,
49  const std::vector<flex_type_enum>& column_types,
50  size_t nrows, Callable next_row) {
51  ASSERT_MSG(column_types.size() == column_names.size(),
52  "column_types size mismatches with column_names size");
53 
54  // calculate intervals
55  size_t nthreads = thread::cpu_count();
56  size_t interval = nrows / nthreads;
57  std::vector<std::pair<size_t, size_t>> write_intervals;
58  for (size_t ii = 0; ii < nthreads; ii++) {
59  if (ii) {
60  auto start = write_intervals.back().second;
61  write_intervals.emplace_back(start, start + interval);
62  } else {
63  write_intervals.emplace_back(0, interval);
64  }
65  }
66  write_intervals.back().second = nrows;
67 
68  // consturct sframe
69  sframe out;
70  out.open_for_write(column_names, column_types, "", nthreads);
71 
72  std::vector<turi::sframe_output_iterator> write_iters;
73  write_iters.resize(nthreads);
74  for (size_t ii = 0; ii < nthreads; ii++)
75  write_iters[ii] = out.get_output_iterator(ii);
76 
77  parallel_for(0, write_iters.size(), [&](size_t ii) {
78  size_t start = write_intervals[ii].first;
79  size_t end = write_intervals[ii].second;
80  auto out_iter = write_iters[ii];
81  /* multithreading: functor may carry a state */
82  /* copy explicitly because next_row might be function pointer */
83  Callable func = next_row;
84  while (start < end) {
85  *out_iter = func(start);
86  start++;
87  }
88  });
89 
90  // finish writing
91  out.close();
92 
93  return out;
94 }

155 template <typename T, typename U>
156 std::vector<T> testing_extract_column(std::shared_ptr<sarray<U> > col) {
157 
158  auto reader = col->get_reader();
159 
160  size_t num_segments = col->num_segments();
161 
162  std::vector<T> ret;
163  ret.reserve(col->size());
164 
165  for(size_t sidx = 0; sidx < num_segments; ++sidx) {
166  auto src_it = reader->begin(sidx);
167  auto src_it_end = reader->end(sidx);
168 
169  for(; src_it != src_it_end; ++src_it)
170  ret.push_back(T(*src_it));
171  }
172 
173  return ret;
174 }

197 // Turn a vector into an sarray (non flexible_type version).
198 template <typename T>
199 std::shared_ptr<sarray<T> > make_testing_sarray(const std::vector<T>& col) {
200  std::shared_ptr<sarray<T> > new_x(new sarray<T>);
201 
202  new_x->open_for_write(1);
203  auto it_out = new_x->get_output_iterator(0);
204 
205  for(const auto& row : col) {
206  *it_out = row;
207  ++it_out;
208  }
209 
210  new_x->close();
211 
212  return new_x;
213 }
