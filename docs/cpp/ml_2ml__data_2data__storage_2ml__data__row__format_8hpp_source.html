17 /**
18  * ml_data layout format
19  * ================================================================================
20  *
21  * The rows are stored in segments of a vector of entry_value structs,
22  * where entry_value encloses a union of an index_value and a
23  * double_value. Thus it is 8 bytes. Each vector contains up to
24  * row_block_size rows. The data is stored as an sarray<vector<entry_value> >
25  *
26  * Raw data layout
27  * ---------------
28  *
29  * Each vector is simply laid out as
30  *
31  * | row 1 | row 2 | ... | row n |
32  *
33  * where n <= row_block_size.
34  *
35  * Each row is laid out according to the "mode" of the columns,
36  * determined by column_metadata::mode. mode can be NUMERIC,
37  * CATEGORICAL, NUMERIC_VECTOR, CATEGORICAL_VECTOR, and7 DICTIONARY. The
38  * last three of these can hold multiple values.
39  *
40  * ROW FORMAT (from ml_data.hpp)
41  * ================================================================================
42  *
43  * ml_data loads data from an existing sframe, indexes it by mapping all
44  * categorical values to unique indices in 0, 1,2,...,n, and records
45  * statistics about the values. It then puts it into an efficient
46  * row-based data storage structure for use in regression and learning
47  * algorithms. The row based storage structure is designed for fast
48  * iteration through the rows and target.
49  *
50  *
51  * The rows are stored in segments of a vector of entry_value structs,
52  * where entry_value encloses a union of an index_value and a
53  * double_value. Thus it is 8 bytes. Each vector contains up to
54  * ROW_BLOCK_SIZE rows. The data is stored as an sarray<vector<entry_value> >
55  *
56  * Raw data layout
57  * ---------------
58  *
59  * Each vector is simply laid out as
60  *
61  * | <row_size> row 1 | <row_size> row 2 | ... | <row_size> row n |
62  *
63  * The row size variable is present if CATEGORICAL_VECTOR or
64  * DICTIONARY are defined types, as these may induce variable sized
65  * rows. Otherwise, the row sizes are constant and determined from
66  * the metadata, thus this does not need to be there.
67  *
68  * where n <= ROW_BLOCK_SIZE.
69  *
70  * Each row is laid out according to the "mode" of the columns,
71  * determined by column_metadata::mode. mode can be NUMERIC,
72  * CATEGORICAL, NUMERIC_VECTOR, CATEGORICAL_VECTOR, and DICTIONARY. The
73  * last three of these can hold multiple values.
74  *
75  * Row layout
76  * ----------
77  *
78  * Each row takes a block of data entries laid out according to mode:
79  *
80  * NUMERIC:
81  *
82  * ... | value | ...
83  *
84  * CATEGORICAL:
85  *
86  * ... | index | ...
87  *
88  * NUMERIC_VECTOR:
89  *
90  * ... | value1 value2 ... | ...
91  *
92  * CATEGORICAL_VECTOR:
93  *
94  * ... | length index1 index2 ... | ...
95  *
96  * DICTIONARY:
97  *
98  * ... | length index1 value1 index2 value2 ... | ...
99  *
100  * example: {0 : 2.1, 1 : 8.5} would be ... | 2 0 2.1 1 8.5 | ...
101  *
102  *
103  * Since the number of columns, and the mode of each column is constant
104  * across rows, the values are all laid out sequentially; for example, a
105  * row with columns (numeric, numeric_vector, dictionary) would be
106  *
107  * ... || row_size | value | v1 v2 ... | length k1 v1 k2 v2 ... || ...
108  *
109  * Thus a row of [1.0, [2.0, 4.0], {0 : 2.1, 1 : 8.5}] would appear as
110  *
111  * ... || 8 | 1.0 | 2.0 4.0 | 2 0 2.1 1 8.5 || ...
112  * = ... || 8 1.0 2.0 4.0 2 0 2.1 1 8.5 || ...
113  *
114  * Similarly, a row with columns (numeric, numeric_vector, categorical)
115  * would be
116  *
117  * ... || value | v1 v2 ... | k1 || ...
118  * * And a row of [1.0, [2.0, 4.0], 8] would appear as
119  *
120  * ... || 1.0 | 2 2.0 4.0 | 8 || ...
121  * = ... || 1.0 2 2.0 4.0 8 || ...
122  * Since ROW_BLOCK_SIZE are stored together in a vector<entry_value>,
123  * iterating through rows is extremely fast.
124  *
125  *
126  * target access
127  * ======================================================================
128  *
129  * If a target column is present, it is stored as the LAST element in
130  * exactly the same way as a regular numeric value.
131  */

173  * block of rows, indexing it through the metadata classes. The

178  *
179  * \param[out] row2data_idx_map A vector corresponding to the number
180  * of rows. Will be giving the starting indices in block_output.
181  *
182  * \param[out] block_output A destination vector of raw row storage.
183  *
184  * \param[in] metadata A vector of column_metadata objects giving the
185  * metadata for each column.
186  *
187  * \param[in] column_buffer A vector of vectors, with each inner
188  * vector holding one unindexed column from the raw source and
189  * corresponding to one of the column_metadata objects. All the
190  * columns must be the same length.
191  *
192  * \param[in] track_statistics If true then the statistics in
193  * metadata is set with the current values.
194  *
195  * \param[in] immutable_metadata If false, then new categorical

199  * categories.
200  *
201  * \param[in] index_remapping If given, map all indices through this

259 /**
260  * Returns the size of the row data block starting at block location
261  * block_location. If has_target is true, the first element is
262  * assumed to be the target value. The rest of the columns are given
263  * by column_modes.
264  *
265  * BlockLocationIterator can be either a pointer or a vector iterator
266  * to the start of the current row.
267  *
268  * The number of data elements in that row is returned.
269  */
270 GL_HOT_INLINE_FLATTEN
271 static inline size_t get_row_data_size(
272  const row_metadata& rm, entry_value_iterator block_location) {
273 
274  if(rm.data_size_is_constant)
275  return rm.constant_data_size;
276  else
277  return block_location->index_value;
278 }

283 GL_HOT_INLINE_FLATTEN

286 
287  if(!rm.has_target || rm.target_is_indexed)

305 /** Reads the rows of data out as a sequence of (column_index,
306  * feature_index, double value) tuples.
307  *
308  * EntryWriteFunction out_function is a function taking 6 arguments:
309  *
310  * (ml_column_mode mode,
311  * size_t column_index,
312  * size_t feature_index,
313  * double value,
314  * size_t index_size,
315  * size_t index_offset).
316  *
317  * row_block_ptr is a pointer or iterator to the first entry_value of
318  * the row.
319  *
320  * rm is the row_metadata structure associated with this row.
321  */
322 template <
323  typename EntryWriteFunction,
324  typename ColumnAdvancementFunction>
325 GL_HOT_INLINE_FLATTEN
326 static inline void
327 read_ml_data_row(
328  const row_metadata& rm,
329  entry_value_iterator& row_block_ptr,
330  EntryWriteFunction&& out_function,
331  ColumnAdvancementFunction&& next_column) {
332 
333 #ifndef NDEBUG
334  entry_value_iterator original_row_block_ptr = row_block_ptr;
335  entry_value_iterator row_block_end = row_block_ptr + get_row_data_size(rm, row_block_ptr);
336 #endif
337 
338  DASSERT_EQ(rm.num_x_columns + (rm.has_target ? 1 : 0), rm.total_num_columns);
339 
340  /** Skip the first entry if the row size is not constant, as this
341  * contains the row data size. */
342  if(!rm.data_size_is_constant)
343  ++row_block_ptr;
344 
345  const size_t num_columns = rm.num_x_columns;
346 
347  for(size_t c_idx = 0; c_idx < num_columns; ++c_idx) {
348 
349  DASSERT_LT(c_idx, rm.metadata_vect.size());
350 
351  ////////////////////////////////////////////////////////////////////////////////
352  // Step 1: Set up everything needed to read out values from the raw data.
353 
354  size_t index_size = rm.metadata_vect[c_idx]->index_size();
355  size_t index_offset = rm.metadata_vect[c_idx]->global_index_offset();
356 
357  // The offset should always have been initialized.
358  DASSERT_NE(index_offset, size_t(-1));
359 
360  /** Call this to read out an index.
361  */
362  auto read_index = [&]() GL_GCC_ONLY(GL_HOT_INLINE_FLATTEN) {
363  DASSERT_LT(row_block_ptr, row_block_end);
364 
365  size_t index = row_block_ptr->index_value;
366  ++row_block_ptr;
367  return index;
368  };
369 
370  /** Call this to read out a value.
371  */
372  auto read_value = [&]() GL_GCC_ONLY(GL_HOT_INLINE_FLATTEN) {
373  DASSERT_LT(row_block_ptr, row_block_end);
374 
375  double value = row_block_ptr->double_value;
376  ++row_block_ptr;
377  return value;
378  };
379 
380  /** Call this to read out a size.
381  */
382  auto read_size = [&]() GL_GCC_ONLY(GL_HOT_INLINE_FLATTEN) {
383  size_t size = row_block_ptr->index_value;
384  ++row_block_ptr;
385 
386  DASSERT_FALSE(rm.data_size_is_constant);
387  DASSERT_LT(size, get_row_data_size(rm, original_row_block_ptr));
388 
389  return size;
390  };
391 
392  ////////////////////////////////////////////////////////////////////////////////
393  // Step 2: Go through and read values based on the type.
394 
395  // For some of the modes, an important optimization is to set
396  // all_columns_categorical to be true in the data profile, which
397  // means that this branch / lookup is optimized out at compile
398  // time.
399 
400  switch(rm.metadata_vect[c_idx]->mode) {
401 
402  case ml_column_mode::NUMERIC: {
403 
404  size_t index = 0;
405  double value = read_value();
406 
407  out_function(ml_column_mode::NUMERIC, c_idx, index, value, index_size, index_offset);
408  next_column(ml_column_mode::NUMERIC, c_idx, index_size);
409 
410  break;
411  }
412 
413  case ml_column_mode::CATEGORICAL: {
414 
415  // Just in case we use the all_columns_categorical override --
416  // check to make sure it's valid.
417  DASSERT_TRUE(rm.metadata_vect[c_idx]->mode == ml_column_mode::CATEGORICAL);
418 
419  size_t index = read_index();
420  double value = 1.0;
421 
422  out_function(ml_column_mode::CATEGORICAL, c_idx, index, value, index_size, index_offset);
423  next_column(ml_column_mode::CATEGORICAL, c_idx, index_size);
424  break;
425  }
426 
427  case ml_column_mode::NUMERIC_VECTOR: {
428 
429  size_t nv = rm.metadata_vect[c_idx]->fixed_column_size();
430 
431  for(size_t k = 0; k < nv; ++k) {
432 
433  size_t index = k;
434  double value = read_value();
435 
436  out_function(ml_column_mode::NUMERIC_VECTOR, c_idx, index, value, index_size, index_offset);
437  }
438 
439  next_column(ml_column_mode::NUMERIC_VECTOR, c_idx, index_size);
440  break;
441  }
442 
443  case ml_column_mode::NUMERIC_ND_VECTOR: {
444 
445  size_t nv = rm.metadata_vect[c_idx]->fixed_column_size();
446 
447  for(size_t k = 0; k < nv; ++k) {
448 
449  size_t index = k;
450  double value = read_value();
451 
452  out_function(ml_column_mode::NUMERIC_ND_VECTOR, c_idx, index, value, index_size, index_offset);
453  }
454 
455  next_column(ml_column_mode::NUMERIC_ND_VECTOR, c_idx, index_size);
456  break;
457  }
458 
459  case ml_column_mode::CATEGORICAL_VECTOR: {
460 
461  size_t nv = read_size();
462 
463  for(size_t k = 0; k < nv; ++k) {
464 
465  size_t index = read_index();
466  double value = 1.0;
467 
468  out_function(ml_column_mode::CATEGORICAL_VECTOR, c_idx, index, value, index_size, index_offset);
469  }
470 
471  next_column(ml_column_mode::CATEGORICAL_VECTOR, c_idx, index_size);
472  break;
473  }
474 
475  case ml_column_mode::DICTIONARY: {
476 
477  size_t nv = read_size();
478 
479  for(size_t k = 0; k < nv; ++k) {
480 
481  size_t index = read_index();
482  double value = read_value();
483 
484  out_function(ml_column_mode::DICTIONARY, c_idx, index, value, index_size, index_offset);
485  }
486 
487  next_column(ml_column_mode::DICTIONARY, c_idx, index_size);
488  break;
489  }
490 
491  case ml_column_mode::UNTRANSLATED: {
492 
493  // Do nothing in this case.
494  break;
495  }
496  default:
497  DASSERT_TRUE(false);
498  } // End switch
499  }
500 }
