47 /**
48  * all_arithmetic<Args...>::value is true if every type in
49  * Args are arithmetic types (numeric).
50  */
51 template <typename... Args>
52 struct all_arithmetic {
53  typedef boost::mpl::vector<Args...> type_sequence;
54  // makes an mpl::vector<true_, false_ ....> where each element transforms
55  // each arg to whether it is flexible_type convertible
56  typedef typename
57  boost::mpl::transform<type_sequence,
58  std::is_arithmetic<boost::mpl::_1>>::type transformed_sequence;
59 
60  // the number of good types (number which are true)
61  typedef typename
62  boost::mpl::count<transformed_sequence,
63  std::integral_constant<bool, true>>::type num_good;
64 
65  // true if all args are convertible.
66  static constexpr bool value = (num_good::value == sizeof...(Args));
67 };

70 /**
71  * \ingroup group_gl_flexible_type
72  * The flexible_type_converter<T> is a class which exposes two member functions.
73  * \code
74  * T flexible_type_converter<T>::get(const flexible_type&)
75  * flexible_type flexible_type<T>::set(const T&)
76  * \endcode
77  *
78  * Essentially flexible_type_converter::get converts from a flexible_type to an
79  * arbitrary type T, and flexible_type_converter::set converts from an
80  * arbitrary type T to a flexible_type.
81  *
82  * The key is to support as interesting types for T as possible.
83  * The following are currently supported:
84  * - flexible_type itself
85  * - flexible_type castable cases:
86  * - Any direct member of flexible_type:
87  * - flex_int (int64_t)
88  * - flex_float (double)
89  * - flex_string (std::string)
90  * - flex_vec (std::vector<double>)
91  * - flex_list (std::vector<flexible_type>)
92  * - flex_dict (std::vector<std::pair<flexible_type, flexible_type> >)
93  * - flex_date_time (boost::ptime)
94  * - Any scalar type (any boolean, integer, or floating point type)
95  * - Recursive cases
96  * - std::vector<T> where T is of any type in this list including the
97  * recursive cases.
98  * - std::map<S, T> where S and T are of any type in this list including
99  * the recursive cases. (though note that complicated S types may not be
100  * representable in Python.)
101  * - std::unordered_map<S, T> where S and T are of any type in this list
102  * including the recursive cases. (though note that complicated S types
103  * may not be representable in Python.)
104  * - std::pair<S, T> where S, T are of any type in this list including the
105  * recursive cases.
106  * - std::tuple<T...> where T... are of any type in this list including the
107  * recursive cases.
108  *
109  * flexible_type_converter_implementation Details
110  * ----------------------
111  * The key difficulty is to make sure that every T matches *exactly* one case.
112  * Each case below is numbered, and we document here how each situation maps
113  * to each case.
114  *
115  * Base case.
116  * All failed templatizations will come here where compilation will fail.
117  */
118 template <typename T, class Enable=void>
119 struct flexible_type_converter {
120  // every successful specialization has a value = True.
121  // This one has a value = False.
122  // This value is used in is_flexible_type_convertible<T> and
123  // all_flexible_type_convertible<Args...> to test if stuff are convertible
124  // to flexible_type via flexible_type_converter<T>
125  static constexpr bool value = false;
126 };

128 /**
129  * Case 1a: Cover all the direct members of flexible_type,
130  * EXCEPT for numeric types (flex_int, flex_float), and flex_list.
131  * - flex_string (std::string)
132  * - flex_vec (std::vector<double>)
133  * - flex_dict (std::vector<std::pair<flexible_type, flexible_type> >)
134  * - flex_date_time (boost::ptime)
135  */
136 template <typename T>
137 struct flexible_type_converter<T,
138  typename std::enable_if<is_flexible_type_member<T>::value &&
139  !std::is_arithmetic<T>::value &&
140  !std::is_same<T, flex_list>::value>::type> {
141  static constexpr bool value = true;
142  T get(const flexible_type& val) {
143  if (type_to_enum<T>::value != val.get_type()) {
144  std::string errormsg = std::string("Expecting ")
145  + flex_type_enum_to_name(type_to_enum<T>::value)
146  + ". But we got a " + flex_type_enum_to_name(val.get_type());
147  throw(errormsg);
148  }
149  return val.get<T>();
150  }
151  flexible_type set(const T& val) {
152  return flexible_type(val);
153  }
154 };

157 /**
158  * Case 1b: flex_list: std::vector<flexible_type>
159  */
160 template <>
161 struct flexible_type_converter<flex_list, void> {
162  static constexpr bool value = true;
163  flex_list get(const flexible_type& val) {
164  if (val.get_type() == flex_type_enum::LIST) {
165  return val;
166  }
167  else if (val.get_type() == flex_type_enum::VECTOR) {
168  flex_vec f = val.to<flex_vec>();
169  flex_list ret;
170  ret.resize(f.size());
171  for (size_t i = 0;i < f.size(); ++i) ret[i] = f[i];
172  return ret;
173  } else {
174  std::string errormsg =
175  std::string("Expecting a list or array, but we got a ")
176  + flex_type_enum_to_name(val.get_type());
177  throw(errormsg);
178  }
179  }
180  flexible_type set(const flex_list& val) {
181  return flexible_type(val);
182  }
183 };

298  typename std::enable_if<(is_flexible_type_convertible<S>::value &&
299  is_flexible_type_convertible<T>::value)>::type> {
300  static constexpr bool value = true;
301  std::map<S, T> get(const flexible_type& val) {
302 
303  if (val.get_type() != flex_type_enum::DICT) {
304  std::string errormsg =
305  std::string("Expecting a dictionary, But we got a ")
306  + flex_type_enum_to_name(val.get_type());
307  throw(errormsg);
308  }
309 
310  const flex_dict& d = val.get<flex_dict>();
311  std::map<S, T> ret;
312  for (const auto& elem: d) {
313  ret.insert({flexible_type_converter<S>().get(elem.first),
314  flexible_type_converter<T>().get(elem.second)});
315  }
316  return ret;
317  }

319  flex_dict ret;
320  for (const auto& elem: val) {
321  ret.push_back({flexible_type_converter<S>().set(elem.first),
322  flexible_type_converter<T>().set(elem.second)});
323  }
324  return flexible_type(ret);
325  }

438 /**
439  * Fills in a tuple<T ...> with values from a flex_list, using
440  * the flexible_type_converter to convert each value.
441  *
442  * operator()<int i> essentially performs
443  * tuple[i] = (convert to tuple_i type)input[i]
444  * where input[i] is a vector of flexible_type
445  */
446 template <typename TupleType>
447 struct fill_tuple_from_flex_list {
448  const flex_list* input;
449  mutable TupleType tuple;
450  template<int n>
451  void operator()(boost::mpl::integral_c<int, n> t) const {
452  typedef typename std::decay<decltype(std::get<n>(tuple))>::type TargetType;
453  std::get<n>(tuple) =
454  flexible_type_converter<TargetType>().get(input->at(n));
455  }
456 };

459 /**
460  * Fills in a flex_list from tuple<T ...>
461  * the flexible_type_converter to convert each value.
462  *
463  * operator()<int i> essentially performs
464  * output[i] = (convert to flexible_type type)tuple[i]
465  * where output[i] is a vector of flexible_type
466  */
467 template <typename TupleType>
468 struct fill_flex_list_from_tuple{
469  const TupleType* input;
470  mutable flex_list output;
471  template<int n>
472  void operator()(boost::mpl::integral_c<int, n> t) const {
473  typedef typename std::decay<decltype(std::get<n>(*input))>::type TargetType;
474  output.at(n) =
475  flexible_type_converter<TargetType>().set(std::get<n>(*input));
476  }
477 };

486  * where input[i] is a double vector

521 /**
522  * Case 10: std::tuple<T...> where T... are all convertible to flexible_type
523  */
524 template <typename... Args>
525 struct flexible_type_converter<std::tuple<Args...>,
526  typename std::enable_if<all_flexible_type_convertible<Args...>::value &&
527  !all_arithmetic<Args...>::value>::type > {
528  static constexpr bool value = true;
529  std::tuple<Args...> get(const flexible_type& val) {
530 
531  if (val.get_type() != flex_type_enum::LIST) {
532  std::string errormsg =
533  std::string("Expecting a list of length ")
534  + std::to_string(sizeof...(Args)) + ", But we got a "
535  + flex_type_enum_to_name(val.get_type());
536  throw(errormsg);
537  }
538  const flex_list& d = val.get<flex_list>();
539  if (d.size() != sizeof...(Args)) {
540  std::string errormsg =
541  std::string("Expecting a list of length ")
542  + std::to_string(sizeof...(Args)) + ", But we got a list of length "
543  + std::to_string(d.size());
544  throw(errormsg);
545  }
546  typename boost::mpl::range_c<int, 0, sizeof...(Args)>::type tuple_range;
547  flexible_type_converter_impl::
548  fill_tuple_from_flex_list<std::tuple<Args...>> filler;
549  filler.input = &d;
550  boost::fusion::for_each(tuple_range, filler);
551  return filler.tuple;
552  }
553  flexible_type set(const std::tuple<Args...> & val) {
554  typename boost::mpl::range_c<int, 0, sizeof...(Args)>::type tuple_range;
555  flexible_type_converter_impl::
556  fill_flex_list_from_tuple<std::tuple<Args...>> filler;
557  filler.input = &val;
558  filler.output.resize(sizeof...(Args));
559  boost::fusion::for_each(tuple_range, filler);
560  return filler.output;
561  }
562 };
