29 template<typename current_pixel_type, typename new_pixel_type>
30 void resize_image_detail(const char* data, size_t width, size_t height, size_t channels, size_t resized_width, size_t resized_height, size_t resized_channels, char** resized_data){
31  if (data == NULL){
32  log_and_throw("Trying to resize image with NULL data pointer");
33  }
34  size_t len = resized_height * resized_width * resized_channels;
35  char* buf = new char[len];
36  // Fast path when the sizes are equal.
37  if ((width == resized_width) && (height == resized_height) && (channels == resized_channels)) {
38  memcpy(buf, data, len);
39  } else {
40  auto view = interleaved_view(width, height, (current_pixel_type*)data, width * channels * sizeof(char));
41  auto resized_view = interleaved_view(resized_width, resized_height, (new_pixel_type*)buf,
42  resized_width * resized_channels * sizeof(char));
43  resize_view(color_converted_view<new_pixel_type>(view), (resized_view), nearest_neighbor_sampler());
44  }
45  *resized_data = buf;
46 }

105 void decode_image_impl(image_type& image) {
106  if (image.m_format == Format::RAW_ARRAY) {
107  return;
108  }
109 
110  char* buf = NULL;
111  size_t length = 0;
112 
113  if (image.m_format == Format::JPG) {
114  decode_jpeg((const char*)image.get_image_data(), image.m_image_data_size,
115  &buf, length);
116  } else if (image.m_format == Format::PNG) {
117  decode_png((const char*)image.get_image_data(), image.m_image_data_size,
118  &buf, length);
119  } else {
120  log_and_throw(std::string("Cannot decode image. Unknown format."));
121  };
122  image.m_image_data.reset(buf);
123  image.m_image_data_size = length;
124  image.m_format = Format::RAW_ARRAY;
125 }
