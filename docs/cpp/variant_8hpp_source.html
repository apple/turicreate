23 /**
24  * A variant object that can be communicated between Python and C++ which
25  * contains either a
26  * \li flexible_type
27  * \li std::shared_ptr<unity_sgraph>
28  * \li dataframe_t
29  * \li model

113 /**
114  * Gets a reference to a content of a variant.
115  * Throws if variant contains an inappropriate type.
116  */
117 template <typename T>
118 inline T& variant_get_ref(variant_type& v) {
119  try {
120  boost::get<T>(v);
121  } catch (...) {
122  std::string errormsg = std::string("Expecting ") +
123  get_variant_which_name(variant_type(T()).which()) +
124  " but got a " + get_variant_which_name(v.which());
125  throw(errormsg);
126  }
127  return boost::get<T>(v);
128 }
