13 /**
14  * \ingroup util
15  * Wrapper around pthread's condition variable, that can work with both
16  * fibers and threads simultaneously, but at a cost of much greater memory
17  * requirements.
18  *
19  * Limitations.
20  * - Does not support timed wait
21  * - threads and fibers are not queued perfectly. fibers are preferentially
22  * signaled.
23  *
24  * \ingroup threading
25  */

52  /// Waits on condition. The mutex must already be acquired. Caller
53  /// must be careful about spurious wakes.
54  inline void wait(const mutex& mut) const {
55  size_t tid = fiber_control::get_tid();
56  if (tid > 0) {
57  lock.lock();
58  fibers.push(tid);
59  lock.unlock();
60  fiber_control::deschedule_self(&mut.m_mut);
61  mut.lock();
62  } else {
63  int error = pthread_cond_wait(&m_cond, &mut.m_mut);
64  ASSERT_TRUE(!error);
65  }
66  }
67  /// Signals one waiting thread to wake up
68  inline void signal() const {
69  if (!fibers.empty()) {
70  lock.lock();
71  if (!fibers.empty()) {
72  size_t tid = fibers.front();
73  fibers.pop();
74  lock.unlock();
75  fiber_control::schedule_tid(tid);
76  return;
77  }
78  lock.unlock();
79  }
80  int error = pthread_cond_signal(&m_cond);
81  ASSERT_TRUE(!error);
82  }
83  /// Wakes up all waiting threads
84  inline void broadcast() const {
85  lock.lock();
86  while (!fibers.empty()) {
87  size_t tid = fibers.front();
88  fibers.pop();
89  fiber_control::schedule_tid(tid);
90  }
91  lock.unlock();
92  int error = pthread_cond_broadcast(&m_cond);
93  ASSERT_TRUE(!error);
94  }
95 
96  ~fiber_conditional() {
97  ASSERT_EQ(fibers.size(), 0);
98  int error = pthread_cond_destroy(&m_cond);
99  ASSERT_TRUE(!error);
100  }
101 };
