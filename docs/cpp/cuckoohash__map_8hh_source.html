77  //! Class returned by operator[] which wraps an entry in the hash table.
78  //! Note that this reference type behave somewhat differently from an STL
79  //! map reference. Most importantly, running this operator will not insert a
80  //! default key-value pair into the map if the given key is not already in
81  //! the map.
82  class reference {
83  // Note that this implementation here is not exactly STL compliant. To
84  // maintain performance and avoid hitting the hash table too many times,
85  // The reference object is *lazy*. In other words,
86  //
87  // - operator[] does not actually perform an insert. It returns a
88  // reference object pointing to the requested key.
89  // - On table[i] = val // reference::operator=(mapped_type)
90  // an update / insert is called
91  // - On table[i] = table[j] // reference::operator=(const reference&)
92  // an update / insert is called with the value of table[j]
93  // - On val = table[i] // operator mapped_type()
94  // a find is called
95  // - On table[i] (i.e. no operation performed)
96  // the destructor is called immediately (reference::~reference())
97  // and nothing happens.
98  public:
99  //! Delete the default constructor, which should never be used
100  reference() = delete;
101 
102  //! Casting to \p mapped_type runs a find for the stored key. If the
103  //! find fails, it will thrown an exception.
104  operator mapped_type() const {
105  return owner_.find(key_);
106  }
107 
108  //! The assignment operator will first try to update the value at the
109  //! reference's key. If the key isn't in the table, it will insert the
110  //! key with \p val.
111  reference& operator=(const mapped_type& val) {
112  owner_.upsert(
113  key_, [&val](mapped_type& v) { v = val; }, val);
114  return *this;
115  }
116 
117  //! The copy assignment operator doesn't actually copy the passed-in
118  //! reference. Instead, it has the same behavior as operator=(const
119  //! mapped_type& val).
120  reference& operator=(const reference& ref) {
121  *this = (mapped_type) ref;
122  return *this;
123  }
124 
125  private:
126  // private constructor which initializes the owner and key
127  reference(
128  cuckoohash_map<Key, T, Hash, Pred, Alloc, SLOT_PER_BUCKET>& owner,
129  const key_type& key) : owner_(owner), key_(key) {}
130 
131  // reference to the hash map instance
132  cuckoohash_map<Key, T, Hash, Pred, Alloc, SLOT_PER_BUCKET>& owner_;
133  // the referenced key
134  const key_type& key_;

351  // A GlobalHazardPointerList stores a list of pointers that cannot be
352  // deleted by an expansion thread. Each thread gets its own node in the
353  // list, whose data pointer it can modify without contention.
354  class GlobalHazardPointerList {
355  std::list<TableInfo*> hp_;
356  std::mutex lock_;
357  public:
358  // new_hazard_pointer creates and returns a new hazard pointer for a
359  // thread.
360  TableInfo** new_hazard_pointer() {
361  std::unique_lock<std::mutex> ul(lock_);
362  hp_.emplace_back(nullptr);
363  return &hp_.back();
364  }
365 
366  // delete_unused scans the list of hazard pointers, deleting any
367  // pointers in old_pointers that aren't in this list. If it does delete
368  // a pointer in old_pointers, it deletes that node from the list.
369  void delete_unused(std::list<std::unique_ptr<TableInfo>>&
370  old_pointers) {
371  std::unique_lock<std::mutex> ul(lock_);
372  old_pointers.remove_if(
373  [this](const std::unique_ptr<TableInfo>& ptr) {
374  return std::find(hp_.begin(), hp_.end(), ptr.get()) ==
375  hp_.end();
376  });
377  }
378  };

636  //! upsert is a combination of update_fn and insert. It first tries updating
637  //! the value associated with \p key using \p fn. If \p key is not in the
638  //! table, then it runs an insert with \p key and \p val. It will always
639  //! succeed, since if the update fails and the insert finds the key already
640  //! inserted, it can retry the update.
641  template <typename Updater>
642  void upsert(const key_type& key, Updater fn, const mapped_type& val) {
643  check_hazard_pointer();
644  check_counterid();
645  size_t hv = hashed_key(key);
646  TableInfo* ti;
647  size_t i1, i2;
648 
649  bool res;
650  do {
651  std::tie(ti, i1, i2) = snapshot_and_lock_two(hv);
652  HazardPointerUnsetter hpu;
653  const cuckoo_status st = cuckoo_update_fn(key, fn, hv, ti, i1, i2);
654  if (st == ok) {

671  //! number of buckets in the table will be 2<SUP>\p n</SUP> after expansion,
672  //! so the table will have 2<SUP>\p n</SUP> &times; \ref SLOT_PER_BUCKET
673  //! slots to store items in. If \p n is not larger than the current
674  //! hashpower, then it decreases the hashpower to either \p n or the
675  //! smallest power that can hold all the elements currently in the table. It

782  // lock_three locks the three bucket indexes in numerical order.
783  static void lock_three(TableInfo* ti, size_t i1,
784  size_t i2, size_t i3) {
785  i1 = lock_ind(i1);
786  i2 = lock_ind(i2);
787  i3 = lock_ind(i3);
788  // If any are the same, we just run lock_two
789  if (i1 == i2) {
790  lock_two(ti, i1, i3);
791  } else if (i2 == i3) {
792  lock_two(ti, i1, i3);
793  } else if (i1 == i3) {
794  lock_two(ti, i1, i2);
795  } else {
796  if (i1 < i2) {
797  if (i2 < i3) {
798  ti->locks_[i1].lock();
799  ti->locks_[i2].lock();
800  ti->locks_[i3].lock();
801  } else if (i1 < i3) {
802  ti->locks_[i1].lock();
803  ti->locks_[i3].lock();
804  ti->locks_[i2].lock();
805  } else {
806  ti->locks_[i3].lock();
807  ti->locks_[i1].lock();
808  ti->locks_[i2].lock();
809  }
810  } else if (i2 < i3) {
811  if (i1 < i3) {
812  ti->locks_[i2].lock();
813  ti->locks_[i1].lock();
814  ti->locks_[i3].lock();
815  } else {
816  ti->locks_[i2].lock();
817  ti->locks_[i3].lock();
818  ti->locks_[i1].lock();
819  }
820  } else {
821  ti->locks_[i3].lock();
822  ti->locks_[i2].lock();
823  ti->locks_[i1].lock();
824  }
825  }

843  // snapshot_table_nolock loads the table info pointer and sets the hazard
844  // pointer, whithout locking anything. There is a possibility that after
845  // loading a snapshot and setting the hazard pointer, an expansion runs and
846  // create a new version of the table, leaving the old one for deletion. To
847  // deal with that, we check that the table_info we loaded is the same as the
848  // current one, and if it isn't, we try again. Whenever we check if (ti !=
849  // table_info.load()) after setting the hazard pointer, there is an ABA
850  // issue, where the address of the new table_info equals the address of a
851  // previously deleted one, however it doesn't matter, since we would still
852  // be looking at the most recent table_info in that case.
853  TableInfo* snapshot_table_nolock() const {
854  while (true) {
855  TableInfo* ti = table_info.load();
856  *hazard_pointer = ti;
857  // If the table info has changed in the time we set the hazard
858  // pointer, ti could have been deleted, so try again.
859  if (ti != table_info.load()) {
860  continue;
861  }
862  return ti;
863  }
864  }

867  // associated with the given hash value. It returns the table_info and the
868  // two locked buckets as a tuple. Since the positions of the bucket locks
869  // depends on the number of buckets in the table, the table_info pointer

1076  // slot_search searches for a cuckoo path using breadth-first search. It
1077  // starts with the i1 and i2 buckets, and, until it finds a bucket with an
1078  // empty slot, adds each slot of the bucket in the b_slot. If the queue runs
1079  // out of space, it fails.
1080  static b_slot slot_search(TableInfo* ti, const size_t i1, const size_t i2) {
1081  b_queue q;
1082  // The initial pathcode informs cuckoopath_search which bucket the path
1083  // starts on
1084  q.enqueue(b_slot(i1, 0, 0));
1085  q.enqueue(b_slot(i2, 1, 0));
1086  while (!q.full() && !q.empty()) {
1087  b_slot x = q.dequeue();
1088  // Picks a (sort-of) random slot to start from
1089  size_t starting_slot = x.pathcode % SLOT_PER_BUCKET;
1090  for (size_t i = 0; i < SLOT_PER_BUCKET && !q.full();
1091  ++i) {
1092  size_t slot = (starting_slot + i) % SLOT_PER_BUCKET;
1093  lock(ti, x.bucket);
1094  if (!ti->buckets_[x.bucket].occupied(slot)) {
1095  // We can terminate the search here
1096  x.pathcode = x.pathcode * SLOT_PER_BUCKET + slot;
1097  unlock(ti, x.bucket);
1098  return x;
1099  }
1100 
1101  // If x has less than the maximum number of path components,
1102  // create a new b_slot item, that represents the bucket we would
1103  // have come from if we kicked out the item at this slot.
1104  if (x.depth < MAX_BFS_PATH_LEN - 1) {
1105  const size_t hv = hashed_key(
1106  ti->buckets_[x.bucket].key(slot));
1107  unlock(ti, x.bucket);
1108  b_slot y(alt_index(ti, hv, x.bucket),
1109  x.pathcode * SLOT_PER_BUCKET + slot, x.depth+1);
1110  q.enqueue(y);
1111  }
1112  }
1113  }
1114  // We didn't find a short-enough cuckoo path, so the queue ran out of
1115  // space. Return a failure value.
1116  return b_slot(0, 0, -1);
1117  }

1119  // cuckoopath_search finds a cuckoo path from one of the starting buckets to
1120  // an empty slot in another bucket. It returns the depth of the discovered
1121  // cuckoo path on success, and -1 on failure. Since it doesn't take locks on
1122  // the buckets it searches, the data can change between this function and
1123  // cuckoopath_move. Thus cuckoopath_move checks that the data matches the
1124  // cuckoo path before changing it.
1125  static int cuckoopath_search(TableInfo* ti, CuckooRecord* cuckoo_path,
1126  const size_t i1, const size_t i2) {
1127  b_slot x = slot_search(ti, i1, i2);
1128  if (x.depth == -1) {
1129  return -1;
1130  }
1131  // Fill in the cuckoo path slots from the end to the beginning
1132  for (int i = x.depth; i >= 0; i--) {
1133  cuckoo_path[i].slot = x.pathcode % SLOT_PER_BUCKET;
1134  x.pathcode /= SLOT_PER_BUCKET;
1135  }
1136  // Fill in the cuckoo_path buckets and keys from the beginning to the
1137  // end, using the final pathcode to figure out which bucket the path
1138  // starts on. Since data could have been modified between slot_search
1139  // and the computation of the cuckoo path, this could be an invalid
1140  // cuckoo_path.
1141  CuckooRecord* curr = cuckoo_path;
1142  if (x.pathcode == 0) {
1143  curr->bucket = i1;
1144  lock(ti, curr->bucket);
1145  if (!ti->buckets_[curr->bucket].occupied(curr->slot)) {
1146  // We can terminate here
1147  unlock(ti, curr->bucket);
1148  return 0;
1149  }
1150  curr->key = ti->buckets_[curr->bucket].key(curr->slot);
1151  unlock(ti, curr->bucket);
1152  } else {
1153  assert(x.pathcode == 1);
1154  curr->bucket = i2;
1155  lock(ti, curr->bucket);
1156  if (!ti->buckets_[curr->bucket].occupied(curr->slot)) {
1157  // We can terminate here
1158  unlock(ti, curr->bucket);
1159  return 0;
1160  }
1161  curr->key = ti->buckets_[curr->bucket].key(curr->slot);
1162  unlock(ti, curr->bucket);
1163  }
1164  for (int i = 1; i <= x.depth; ++i) {
1165  CuckooRecord* prev = curr++;
1166  const size_t prevhv = hashed_key(prev->key);
1167  assert(prev->bucket == index_hash(ti, prevhv) ||
1168  prev->bucket == alt_index(ti, prevhv, index_hash(ti,
1169  prevhv)));
1170  // We get the bucket that this slot is on by computing the alternate
1171  // index of the previous bucket
1172  curr->bucket = alt_index(ti, prevhv, prev->bucket);
1173  lock(ti, curr->bucket);
1174  if (!ti->buckets_[curr->bucket].occupied(curr->slot)) {
1175  // We can terminate here
1176  unlock(ti, curr->bucket);
1177  return i;
1178  }
1179  curr->key = ti->buckets_[curr->bucket].key(curr->slot);
1180  unlock(ti, curr->bucket);
1181  }
1182  return x.depth;
1183  }

1186  // cuckoopath_move moves keys along the given cuckoo path in order to make
1187  // an empty slot in one of the buckets in cuckoo_insert. Before the start of
1188  // this function, the two insert-locked buckets were unlocked in run_cuckoo.
1189  // At the end of the function, if the function returns true (success), then
1190  // the last bucket it looks at (which is either i1 or i2 in run_cuckoo)
1191  // remains locked. If the function is unsuccessful, then both insert-locked
1192  // buckets will be unlocked.
1193  static bool cuckoopath_move(
1194  TableInfo* ti, CuckooRecord* cuckoo_path, size_t depth,
1195  const size_t i1, const size_t i2) {
1196  if (depth == 0) {
1197  // There is a chance that depth == 0, when try_add_to_bucket sees i1
1198  // and i2 as full and cuckoopath_search finds one empty. In this
1199  // case, we lock both buckets. If the bucket that cuckoopath_search
1200  // found empty isn't empty anymore, we unlock them and return false.
1201  // Otherwise, the bucket is empty and insertable, so we hold the
1202  // locks and return true.
1203  const size_t bucket = cuckoo_path[0].bucket;
1204  assert(bucket == i1 || bucket == i2);
1205  lock_two(ti, i1, i2);
1206  if (!ti->buckets_[bucket].occupied(cuckoo_path[0].slot)) {
1207  return true;
1208  } else {
1209  unlock_two(ti, i1, i2);
1210  return false;
1211  }
1212  }
1213 
1214  while (depth > 0) {
1215  CuckooRecord* from = cuckoo_path + depth - 1;
1216  CuckooRecord* to = cuckoo_path + depth;
1217  size_t fb = from->bucket;
1218  size_t fs = from->slot;
1219  size_t tb = to->bucket;
1220  size_t ts = to->slot;
1221 
1222  size_t ob = 0;
1223  if (depth == 1) {
1224  // Even though we are only swapping out of i1 or i2, we have to
1225  // lock both of them along with the slot we are swapping to,
1226  // since at the end of this function, i1 and i2 must be locked.
1227  ob = (fb == i1) ? i2 : i1;
1228  lock_three(ti, fb, tb, ob);
1229  } else {
1230  lock_two(ti, fb, tb);
1231  }
1232 
1233  // We plan to kick out fs, but let's check if it is still there;
1234  // there's a small chance we've gotten scooped by a later cuckoo. If
1235  // that happened, just... try again. Also the slot we are filling in
1236  // may have already been filled in by another thread, or the slot we
1237  // are moving from may be empty, both of which invalidate the swap.
1238  if (!eqfn()(ti->buckets_[fb].key(fs), from->key) ||
1239  ti->buckets_[tb].occupied(ts) ||
1240  !ti->buckets_[fb].occupied(fs)) {
1241  if (depth == 1) {
1242  unlock_three(ti, fb, tb, ob);
1243  } else {
1244  unlock_two(ti, fb, tb);
1245  }
1246  return false;
1247  }
1248 
1249  if (!is_simple) {
1250  ti->buckets_[tb].partial(ts) = ti->buckets_[fb].partial(fs);
1251  }
1252  ti->buckets_[tb].setKV(ts, ti->buckets_[fb].key(fs),
1253  std::move(ti->buckets_[fb].val(fs)));
1254  ti->buckets_[fb].eraseKV(fs);
1255  if (depth == 1) {
1256  // Don't unlock fb or ob, since they are needed in
1257  // cuckoo_insert. Only unlock tb if it doesn't unlock the same
1258  // bucket as fb or ob.
1259  if (lock_ind(tb) != lock_ind(fb) &&
1260  lock_ind(tb) != lock_ind(ob)) {
1261  unlock(ti, tb);
1262  }
1263  } else {
1264  unlock_two(ti, fb, tb);
1265  }
1266  depth--;
1267  }
1268  return true;
1269  }

1271  // run_cuckoo performs cuckoo hashing on the table in an attempt to free up
1272  // a slot on either i1 or i2. On success, the bucket and slot that was freed
1273  // up is stored in insert_bucket and insert_slot. In order to perform the
1274  // search and the swaps, it has to unlock both i1 and i2, which can lead to
1275  // certain concurrency issues, the details of which are explained in the
1276  // function. If run_cuckoo returns ok (success), then the slot it freed up
1277  // is still locked. Otherwise it is unlocked.
1278  cuckoo_status run_cuckoo(TableInfo* ti, const size_t i1, const size_t i2,
1279  size_t &insert_bucket, size_t &insert_slot) {
1280 
1281  CuckooRecord cuckoo_path[MAX_BFS_PATH_LEN];
1282 
1283  // We must unlock i1 and i2 here, so that cuckoopath_search and
1284  // cuckoopath_move can lock buckets as desired without deadlock.
1285  // cuckoopath_move has to look at either i1 or i2 as its last slot, and
1286  // it will lock both buckets and leave them locked after finishing. This
1287  // way, we know that if cuckoopath_move succeeds, then the buckets
1288  // needed for insertion are still locked. If cuckoopath_move fails, the
1289  // buckets are unlocked and we try again. This unlocking does present
1290  // two problems. The first is that another insert on the same key runs
1291  // and, finding that the key isn't in the table, inserts the key into
1292  // the table. Then we insert the key into the table, causing a
1293  // duplication. To check for this, we search i1 and i2 for the key we
1294  // are trying to insert before doing so (this is done in cuckoo_insert,
1295  // and requires that both i1 and i2 are locked). Another problem is that
1296  // an expansion runs and changes table_info, meaning the cuckoopath_move
1297  // and cuckoo_insert would have operated on an old version of the table,
1298  // so the insert would be invalid. For this, we check that ti ==
1299  // table_info.load() after cuckoopath_move, signaling to the outer
1300  // insert to try again if the comparison fails.
1301  unlock_two(ti, i1, i2);
1302 
1303  bool done = false;
1304  while (!done) {
1305  int depth = cuckoopath_search(ti, cuckoo_path, i1, i2);
1306  if (depth < 0) {
1307  break;
1308  }
1309 
1310  if (cuckoopath_move(ti, cuckoo_path, depth, i1, i2)) {
1311  insert_bucket = cuckoo_path[0].bucket;
1312  insert_slot = cuckoo_path[0].slot;
1313  assert(insert_bucket == i1 || insert_bucket == i2);
1314  assert(!ti->locks_[lock_ind(i1)].try_lock());
1315  assert(!ti->locks_[lock_ind(i2)].try_lock());
1316  assert(!ti->buckets_[insert_bucket].occupied(insert_slot));
1317  done = true;
1318  break;
1319  }
1320  }
1321 
1322  if (!done) {
1323  return failure;
1324  } else if (ti != table_info.load()) {
1325  // Unlock i1 and i2 and signal to cuckoo_insert to try again. Since
1326  // we set the hazard pointer to be ti, this check isn't susceptible
1327  // to an ABA issue, since a new pointer can't have the same address
1328  // as ti.
1329  unlock_two(ti, i1, i2);
1330  return failure_under_expansion;
1331  }
1332  return ok;
1333  }

1509  // cuckoo_insert tries to insert the given key-value pair into an empty slot
1510  // in i1 or i2, performing cuckoo hashing if necessary. It expects the locks
1511  // to be taken outside the function, but they are released here, since
1512  // different scenarios require different handling of the locks. Before
1513  // inserting, it checks that the key isn't already in the table. cuckoo
1514  // hashing presents multiple concurrency issues, which are explained in the
1515  // function.
1516  template <class V>
1517  cuckoo_status cuckoo_insert(const key_type &key, V val,
1518  const size_t hv, TableInfo* ti,
1519  const size_t i1, const size_t i2) {
1520  int res1, res2;
1521  const partial_t partial = partial_key(hv);
1522  if (!try_find_insert_bucket(ti, partial, key, i1, res1)) {
1523  unlock_two(ti, i1, i2);
1524  return failure_key_duplicated;
1525  }
1526  if (!try_find_insert_bucket(ti, partial, key, i2, res2)) {
1527  unlock_two(ti, i1, i2);
1528  return failure_key_duplicated;
1529  }
1530  if (res1 != -1) {
1531  add_to_bucket(ti, partial, key, std::forward<V>(val), i1, res1);
1532  unlock_two(ti, i1, i2);
1533  return ok;
1534  }
1535  if (res2 != -1) {
1536  add_to_bucket(ti, partial, key, std::forward<V>(val), i2, res2);
1537  unlock_two(ti, i1, i2);
1538  return ok;
1539  }
1540 
1541  // we are unlucky, so let's perform cuckoo hashing
1542  size_t insert_bucket = 0;
1543  size_t insert_slot = 0;
1544  cuckoo_status st = run_cuckoo(ti, i1, i2, insert_bucket, insert_slot);
1545  if (st == failure_under_expansion) {
1546  // The run_cuckoo operation operated on an old version of the table,
1547  // so we have to try again. We signal to the calling insert method
1548  // to try again by returning failure_under_expansion.
1549  return failure_under_expansion;
1550  } else if (st == ok) {
1551  assert(!ti->locks_[lock_ind(i1)].try_lock());
1552  assert(!ti->locks_[lock_ind(i2)].try_lock());
1553  assert(!ti->buckets_[insert_bucket].occupied(insert_slot));
1554  assert(insert_bucket == index_hash(ti, hv) ||
1555  insert_bucket == alt_index(ti, hv, index_hash(ti, hv)));
1556  // Since we unlocked the buckets during run_cuckoo, another insert
1557  // could have inserted the same key into either i1 or i2, so we
1558  // check for that before doing the insert.
1559  if (cuckoo_contains(key, hv, ti, i1, i2)) {
1560  unlock_two(ti, i1, i2);
1561  return failure_key_duplicated;
1562  }
1563  add_to_bucket(ti, partial, key, std::forward<V>(val),
1564  insert_bucket, insert_slot);
1565  unlock_two(ti, i1, i2);
1566  return ok;
1567  }
1568  assert(st == failure);
1569  LIBCUCKOO_DBG("hash table is full (hashpower = %zu, hash_items = %zu,"
1570  "load factor = %.2f), need to increase hashpower\n",
1571  ti->hashpower_, cuckoo_size(ti), cuckoo_loadfactor(ti));
1572  return failure_table_full;
1573  }

1575  // We run cuckoo_insert in a loop until it succeeds in insert and upsert, so
1576  // we pulled out the loop to avoid duplicating it. This should be called
1577  // directly after snapshot_and_lock_two, and by the end of the function, the
1578  // hazard pointer will have been unset.
1579  template <class V>
1580  bool cuckoo_insert_loop(const key_type& key, V val,
1581  size_t hv, TableInfo* ti, size_t i1, size_t i2) {
1582  cuckoo_status st = cuckoo_insert(key, std::forward<V>(val),
1583  hv, ti, i1, i2);
1584  while (st != ok) {
1585  // If the insert failed with failure_key_duplicated, it returns here
1586  if (st == failure_key_duplicated) {
1587  return false;
1588  }
1589  // If it failed with failure_under_expansion, the insert operated on
1590  // an old version of the table, so we just try again. If it's
1591  // failure_table_full, we have to expand the table before trying
1592  // again.
1593  if (st == failure_table_full) {
1594  if (cuckoo_expand_simple(ti->hashpower_ + 1, true) ==
1595  failure_under_expansion) {
1596  LIBCUCKOO_DBG("expansion is on-going\n");
1597  }
1598  }
1599  std::tie(ti, i1, i2) = snapshot_and_lock_two(hv);
1600  st = cuckoo_insert(key, std::forward<V>(val), hv, ti, i1, i2);
1601  }
1602  return true;
1603  }

1703  // cuckoo_expand_simple will resize the table to at least the given
1704  // new_hashpower. If is_expansion is true, new_hashpower must be greater
1705  // than the current size of the table. If it's false, then new_hashpower
1706  // must be less. When we're shrinking the table, if the current table
1707  // contains more elements than can be held by new_hashpower, the resulting
1708  // hashpower will be greater than new_hashpower. It needs to take all the
1709  // bucket locks, since no other operations can change the table during
1710  // expansion.
1711  cuckoo_status cuckoo_expand_simple(size_t new_hashpower,
1712  bool is_expansion) {
1713  TableInfo* ti = snapshot_and_lock_all();
1714  assert(ti == table_info.load());
1715  AllUnlocker au(ti);
1716  HazardPointerUnsetter hpu;
1717  if ((is_expansion && new_hashpower <= ti->hashpower_) ||
1718  (!is_expansion && new_hashpower >= ti->hashpower_)) {
1719  // Most likely another expansion ran before this one could grab the
1720  // locks
1721  return failure_under_expansion;
1722  }
1723 
1724  // Creates a new hash table with hashpower new_hashpower and adds all
1725  // the elements from the old buckets
1726  cuckoohash_map<Key, T, Hash, Pred, Alloc, SLOT_PER_BUCKET> new_map(
1727  hashsize(new_hashpower) * SLOT_PER_BUCKET);
1728  const size_t threadnum = kNumCores();
1729  const size_t buckets_per_thread =
1730  hashsize(ti->hashpower_) / threadnum;
1731  std::vector<std::thread> insertion_threads(threadnum);
1732  for (size_t i = 0; i < threadnum-1; ++i) {
1733  insertion_threads[i] = std::thread(
1734  insert_into_table, std::ref(new_map),
1735  ti, i*buckets_per_thread, (i+1)*buckets_per_thread);
1736  }
1737  insertion_threads[threadnum-1] = std::thread(
1738  insert_into_table, std::ref(new_map), ti,
1739  (threadnum-1)*buckets_per_thread, hashsize(ti->hashpower_));
1740  for (size_t i = 0; i < threadnum; ++i) {
1741  insertion_threads[i].join();
1742  }
1743  // Sets this table_info to new_map's. It then sets new_map's
1744  // table_info to nullptr, so that it doesn't get deleted when
1745  // new_map goes out of scope
1746  table_info.store(new_map.table_info.load());
1747  new_map.table_info.store(nullptr);
