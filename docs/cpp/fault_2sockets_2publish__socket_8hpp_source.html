23 /**
24  * \ingroup fault
25  * Constructs a zookeeper backed publish socket.
26  * This object is very much single threaded.
27  * Sends to this socket will be received by all machines
28  * subscribed to the socket.
29  *
30  * \code
31  * publish_socket pubsock(zmq_ctx, NULL, listen_addr);
32  * pubsock.send(...);
33  * \endcode
34  */
35 class EXPORT publish_socket {
36  public:
37  /**
38  * Constructs a request socket.
39  * The request will be sent to the current owners of the key
40  *
41  * \param zmq_ctx A zeroMQ Context
42  * \param keyval A zookeeper key_value object to bind to
43  * \param alternate_bind_address If set, this will be address to bind to.
44  * Otherwise, binds to a free tcp address.
45  *
46  * keyval can be NULL, in which case, the alternate_bind_address must be
47  * provided, in which case this socket behaves like a simple
48  * messaging wrapper around ZeroMQ which provides a publish socket.
49  */
50  publish_socket(void* zmq_ctx,
51  turi::zookeeper_util::key_value* keyval,
52  std::string alternate_bind_address = "");
53 
54  /**
55  * Closes this socket. Once closed, the socket cannot be used again.
56  */
57  void close();
58 
59 
60  /**
61  * Sends a message. All subscribers which match the message (by prefix)
62  * will receive a copy.
63  */
64  void send(zmq_msg_vector& msg);
65 
66  ~publish_socket();
67 
68  /**
69  * Tries to register this socket under a given object key.
70  * Returns true on success and false on failure. Only relevant when zookeeper
71  * is used. Otherwise this will always return true.
72  */
73  bool register_key(std::string key);
74 
75  /**
76  * Like register, but sets the key to an empty value.
77  * Reserves ownership of the key, but prohibits people from joining. Only
78  * relevant when zookeeper is used. Otherwise this will always return true.
79  */
80  bool reserve_key(std::string key);
81 
82 
83  /**
84  * Tries to unregister this socket from a given object key.
85  * Returns true on success and false on failure. Only relevant when zookeeper
86  * is used. Otherwise this will always return true.
87  */
88  bool unregister_key(std::string key);
89 
90  /**
91  * Unregisters all keys this socket was registered under. Only relevant when
92  * zookeeper is used. Otherwise this will always return true.
93  */
94  void unregister_all_keys();
95 
96  /**
97  * Returns the address the socket is bound to
98  */
99  std::string get_bound_address();
100 
101  private:
