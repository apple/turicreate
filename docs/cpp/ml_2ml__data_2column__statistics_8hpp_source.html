58  /* The count; index here is the index obtained by one of the

61  size_t count(size_t index=0) const {
62  // Use this to
63  if(mode == ml_column_mode::NUMERIC
64  || mode == ml_column_mode::NUMERIC_VECTOR) {
65

74  /* The mean; index here is the index obtained by one of the
75  * map_value_to_index functions previously.
76  */
77  double mean(size_t index=0) const {
78  if(mode == ml_column_mode::CATEGORICAL
79  || mode == ml_column_mode::CATEGORICAL_VECTOR) {
80 
81  double p = double(count(index)) / (std::max(1.0, double(total_row_count)));
82 
83  DASSERT_LE(p, 1.0);
84  DASSERT_GE(p, 0.0);
85 
86  return p;
87 
88  } else {
89 
90  if(total_row_count)
91  DASSERT_TRUE(!statistics.empty());
92 
93  return index < statistics.size() ? statistics[index].mean : 0;
94  }
95  }
96 
97  /* The variance; index here is the index obtained by one of the
98  * map_value_to_index functions previously.
99  */
100  double stdev(size_t index=0) const {
101  if(mode == ml_column_mode::CATEGORICAL
102  || mode == ml_column_mode::CATEGORICAL_VECTOR) {
103 
104  double stdev = 0;
105  double p = mean(index);
106 
107  DASSERT_LE(p, 1.0);
108  DASSERT_GE(p, 0.0);
109 
110  if (total_row_count > 1) {
111  stdev = std::sqrt(total_row_count * p * (1 - p) / (total_row_count-1));
112  } else {
113  stdev = 0;
114  }
115 
116  DASSERT_FALSE(std::isnan(stdev));
117 
118  return stdev;
119 
120  } else {
121 
122  if(total_row_count) {
123  DASSERT_TRUE(!statistics.empty());
124  }
125 
126  return index < statistics.size() ? statistics[index].stdev : 0;
127  }
128  }

137  /// Update categorical statistics for a batch of categorical indices.
138  void update_categorical_statistics(size_t thread_idx, const std::vector<size_t>& cat_index_vect) GL_HOT {

149

151  cv_idx < cat_index_vect.size()
152  && cat_index_vect[cv_idx] < parallel_threshhold;
153  ++cv_idx) {

162  }

173  ++(global_element_counts[idx]);
174  }
175  }
176  }
177 
178  ++by_thread_row_counts[thread_idx];

181  /// Update categorical statistics for a batch of real values.
182  void update_numeric_statistics(size_t thread_idx, const std::vector<double>& value_vect) GL_HOT {
183 
184  DASSERT_TRUE(mode == ml_column_mode::NUMERIC
185  || mode == ml_column_mode::NUMERIC_VECTOR
186  || mode == ml_column_mode::NUMERIC_ND_VECTOR);
187 
188  // Silently ignore columns of empty vectors. Note that all the
189  // vectors in a column must be empty for this to work.
190  if(value_vect.empty()) {
191  return;
192  }
193 
194  // Numeric statistics are always cached on a thread-local basis
195  // and ignore the parallel_threshhold parameter.
196  size_t& count = by_thread_row_counts[thread_idx];
197  auto& stats = by_thread_mean_var_acc[thread_idx];
198 
199  if(stats.empty()) {
200 
201  DASSERT_EQ(size_t(count), 0);
202 
203  stats.resize(value_vect.size());
204 
205  for(size_t i = 0; i < value_vect.size(); ++i) {
206  stats[i].mean = value_vect[i];
207  stats[i].var_sum = 0;
208  }
209 
210  } else {
211  DASSERT_EQ(stats.size(), value_vect.size());
212 
213  // Efficient and Stable Mean+Stdev computation
214  // --------------------------------------------------------------------
215  // This way of computing stdev goes back to a 1962 paper by B. P.
216  // Welford and is presented in Donald Knuth's Art of Computer
217  // Programming, Vol 2, page 232, 3rd edition. It is way more accurate
218  // and stable than computing it "directly".
219  //
220  // Online update rule:
221  //
222  // M_k = M_{k-1} + (x_k - M_{k-1})/k
223  // S_k = S_{k-1} + (x_k - M_{k-1})*(x_k - M_k).
224  //
225  // Here: M_k is the estimate of the mean and S_k/(k-1) is the estimate
226  // of the variance:
227  //
228  for(size_t i = 0; i < value_vect.size(); i++){
229  double& mean = stats[i].mean;
230  double& var_sum = stats[i].var_sum;
231 
232  double old_mean = mean;
233  double v = value_vect[i];
234 
235  mean += (v - old_mean) / (count + 1);
236 
237  // stdev here is actually the variance times the counts; it's
238  // converted to the stdev later
239  var_sum += (v - old_mean) * (v - mean);
240  }
241  }
242 
243  ++count;
244  }

246  /// Update statistics after observing a dictionary.
247  void update_dict_statistics(size_t thread_idx, const std::vector<std::pair<size_t, double> >& dict) GL_HOT {
248 
249  DASSERT_TRUE(mode == ml_column_mode::DICTIONARY);
250 
251  // Does not compile on windows; comment out for now
252  // // The input array is sorted here.
253  // DASSERT_TRUE(std::is_sorted(dict.begin(), dict.end(),
254  // [](std::pair<size_t, double> p1, std::pair<size_t, double> p2) {
255  // return p1.first < p2.first;
256  // }));
257 
258  size_t& row_count = by_thread_row_counts[thread_idx];
259 
260  // The update function for all this
261  auto update_f = [&](size_t& count, double& mean, double& var_sum, double v) {
262 
263  // Efficient and Stable Mean+Stdev computation on Sparse data
264  // --------------------------------------------------------------------
265  // We combine means and variances of 2 samples as follows. The
266  // first sample contains all the non-zeros while the second
267  // contains only zeros.
268  //
269  // M_{a+b} = m * M_a / (m + n)
270  // S_{a+b} = S_{a} + M_a^2 - M_{a+b}^2 * (a+b)
271  //
272  // This computation is O(m) (insted of being O(n+m)).
273  //
274 
275  if(count == 0) {
276  count = 1;
277  mean = v;
278  var_sum = 0;
279 
280  } else {
281  double old_mean = mean;
282 
283  ++count;
284  mean += (v - old_mean) / count;
285  var_sum += (v - old_mean) * (v - mean);
286  }
287  };
288 
289  // First, go through and work with the common elements.
290  std::vector<size_t>& counts = by_thread_element_counts[thread_idx];
291  std::vector<element_statistics_accumulator>& stats = by_thread_mean_var_acc[thread_idx];
292 
293  size_t d_idx = 0;
294 
295  for(; d_idx < dict.size() && dict[d_idx].first < parallel_threshhold; ++d_idx) {
296  size_t idx = dict[d_idx].first;
297  double v = dict[d_idx].second;
298 
299  check_local_array_size(idx, counts);
300  check_local_array_size(idx, stats);
301 
302  update_f(counts[idx], stats[idx].mean, stats[idx].var_sum, v);
303  }
304 
305  // Okay, there are infrequent elements here, so we need to look at the global update deal.
306  if(d_idx != dict.size()) {
307 
308  for(; d_idx < dict.size(); ++d_idx) {
309 
310  size_t idx = dict[d_idx].first - parallel_threshhold;
311  double v = dict[d_idx].second;
312 
313  if(idx >= global_element_counts.size()
314  || idx >= global_mean_var_acc.size()
315  || idx >= global_size) {
316  adjust_global_array_size(idx, global_element_counts);
317  adjust_global_array_size(idx, global_mean_var_acc);
318  }
319 
320  std::lock_guard<simple_spinlock> el_lg(global_element_locks[get_lock_index(idx)]);
321 
322  update_f(global_element_counts[idx],
323  global_mean_var_acc[idx].mean,
324  global_mean_var_acc[idx].var_sum, v);
325  }
326  }
327 
328  ++row_count;
329  }

452  /** Possibly resize the local array.
453  */
454  template <typename T>
455  inline void check_local_array_size(size_t idx, std::vector<T>& v) {
456  DASSERT_LT(idx, parallel_threshhold);
457 
458  // See if a resize is needed.
459  if(idx >= v.size() ) {
460 
461  if(UNLIKELY(v.capacity() < idx + 1)) {
462  size_t new_capacity = std::min(parallel_threshhold, 3 * (idx + 1) / 2);
463 
464  // If it's likely to go to max capacity, just make it that to avoid future resizes.
465  if(new_capacity > parallel_threshhold / 2)
466  new_capacity = parallel_threshhold;
467 
468  v.reserve(new_capacity);
469  }
470 
471  v.resize(idx + 1);
472  }
473  }
474 
475  /** Check global array size. Possibly resize it.
476  */
477  template <typename T>
478  void adjust_global_array_size(size_t idx, std::vector<T>& v) {
479 
480  // If needed, increase the value of global_size.
481  atomic_set_max(global_size, idx + 1);
482 
483  // See if a resize is needed.
484  if(UNLIKELY(idx >= v.size() )) {
485 
486  // Grow aggressively, since a resize is really expensive.
487  size_t new_size = 2 * (parallel_threshhold + idx + 1);
488 
489  {
490  std::array<std::unique_lock<simple_spinlock>, n_locks> all_locks;
491  for(size_t i = 0; i < n_locks; ++i)
492  all_locks[i] = std::unique_lock<simple_spinlock>(global_element_locks[i], std::defer_lock);
493 
494  // Ensure nothing is happening with the vector by locking all
495  // locks in a thread safe way. This prevents any thread from
496  // accessing it while we resize it.
497  boost::lock(all_locks.begin(), all_locks.end());
498 
499  // It's possible that another thread beat us to it.
500  if(v.size() < idx + 1)
501  v.resize(new_size);
502 
503  // The destructor of all_locks takes care of the unlocking.
504  }
505  }
506  }
