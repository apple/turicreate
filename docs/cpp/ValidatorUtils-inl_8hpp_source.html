17  /*
18  * Utility that make sures the feature types are valid.
19  *
20  * @param allowedFeatureTypes Allowed feature types.
21  * @param featureType type of this operation.
22  * @retun
23  */
24  inline Result validateSchemaTypes(const std::vector<Specification::FeatureType::TypeCase>& allowedFeatureTypes,
25  const Specification::FeatureDescription& featureDesc) {
26 
27  // Check the types
28  auto type = featureDesc.type().Type_case();
29  for (const auto& t : allowedFeatureTypes) {
30  if (type == t) {
31  // no invariant broken -- type matches one of the allowed types
32  return Result();
33  }
34  }
35 
36  // Invalid type
37  std::stringstream out;
38  out << "Unsupported type \"" << MLFeatureTypeType_Name(static_cast<MLFeatureTypeType>(featureDesc.type().Type_case()))
39  << "\" for feature \"" << featureDesc.name() + "\". Should be one of: ";
40  bool isFirst = true;
41  for (const auto& t: allowedFeatureTypes) {
42  if (!isFirst) {
43  out << ", ";
44  }
45  out << MLFeatureTypeType_Name(static_cast<MLFeatureTypeType>(t));
46  isFirst = false;
47  }
48  out << "." << std::endl;
49  return Result(ResultType::UNSUPPORTED_FEATURE_TYPE_FOR_MODEL_TYPE, out.str());
50  }

55  template <typename Descriptions>

57  Result result;
58  for (int i = 0; i < features.size(); i++) {
59  result = validateSchemaTypes({
60  Specification::FeatureType::kDoubleType,
61  Specification::FeatureType::kInt64Type,
62  Specification::FeatureType::kMultiArrayType,

65  return result;
66  }
67  }

72  /*
73  * Utility that checks a set of descriptions to validate
74  * there is a feature with a specific name and type in an allowed set
75  */
76  template <typename Descriptions>
77  inline Result validateDescriptionsContainFeatureWithTypes(const Descriptions &features,
78  int maxFeatureCount,
79  const std::vector<Specification::FeatureType::TypeCase>& allowedFeatureTypes) {
80  Result result;
81 
82  // 0 means no maximum fixed feature count.
83  if (maxFeatureCount != 0 && features.size() > maxFeatureCount) {
84  return Result(ResultType::TOO_MANY_FEATURES_FOR_MODEL_TYPE, "Feature descriptions exceeded " + std::to_string(maxFeatureCount));
85  }
86 
87  for (int i = 0; i < features.size(); i++) {
88  result = validateSchemaTypes(allowedFeatureTypes, features[i]);
89  if (!result.good()) {
90  return result;
91  }
92  }
93 
94  return result;
95  }

98  * Utility that checks a set of descriptions to validate
99  * there is a feature with a specific name and type in an allowed set
100  */
101  template <typename Descriptions>
102  inline Result validateDescriptionsContainFeatureWithNameAndType(const Descriptions &features,
103  const std::string &name,
104  const std::vector<Specification::FeatureType::TypeCase>& allowedFeatureTypes) {
105  Result result;
106  for (int i = 0; i < features.size(); i++) {
107  if (name.compare(features[i].name()) != 0) {
108  continue;
109  }
110  return validateSchemaTypes(allowedFeatureTypes, features[i]);
111  }
112 
113  return Result(ResultType::INTERFACE_FEATURE_NAME_MISMATCH, "Expected feature '" + name + "' to the model is not present in the model description.");
114  }
